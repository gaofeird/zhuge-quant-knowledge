# 量化数据工具集成方案

## 一、项目概述

本方案整合 **MinerU**（文档解析）与 **Crawlee**（智能爬虫）两大开源工具，结合已安装的量化技能，构建完整的量化数据工具链。

### 1.1 核心工具简介

| 工具 | GitHub | Stars | 定位 | 核心能力 |
|------|--------|-------|------|----------|
| **MinerU** | [opendatalab/MinerU](https://github.com/opendatalab/MinerU) | 59k ⭐ | 文档解析引擎 | PDF/Word/PPT → Markdown/JSON，表格还原、公式识别 |
| **Crawlee** | [apify/crawlee-python](https://github.com/apify/crawlee-python) | 5.8k ⭐ | 智能爬虫框架 | TLS指纹伪装、代理轮换、反爬强 |

#### 2.3.3 策略文档整理 → 知识库构建

```python
"""
量化策略文档知识库构建
将策略文档、研报、书籍转为可检索的知识库
"""
from mineru import MinerU
from langchain.text_splitter import RecursiveCharacterTextSplitter
from langchain.embeddings import OpenAIEmbeddings
from langchain.vectorstores import Chroma

parser = MinerU(backend="pipeline")

# 批量解析策略文档
docs = []
for pdf_file in Path("./策略文档库").glob("*.pdf"):
    result = parser.parse(pdf_file)
    docs.append({
        "content": result.content["text"],
        "metadata": {
            "source": str(pdf_file),
            "type": "策略文档"
        }
    })

# 文档分块
text_splitter = RecursiveCharacterTextSplitter(
    chunk_size=1000,
    chunk_overlap=200
)
splits = text_splitter.split_documents(docs)

# 向量化存储
vectorstore = Chroma.from_documents(
    documents=splits,
    embedding=OpenAIEmbeddings()
)

# RAG 检索示例
def query_strategy(query: str):
    """查询相关策略"""
    docs = vectorstore.similarity_search(query, k=3)
    return "\n\n".join([d.page_content for d in docs])
```

```python
"""
财经新闻舆情采集
采集东方财富、同花顺等财经网站的新闻和评论
"""
import asyncio
from crawlee.crawlers import PlaywrightCrawler, PlaywrightCrawlingContext
from datetime import datetime

class FinanceNewsCrawler(PlaywrightCrawler):
    """财经新闻爬虫"""

    def __init__(self):
        super().__init__(
            max_requests_per_crawl=50,
            headless=True,
            browser_type='chromium',
        )
        self.news_data = []

    @self.router.default_handler
    async def handle_news(self, context: PlaywrightCrawlingContext):
        # 等待内容加载
        await context.page.wait_for_selector('.news-item', timeout=10000)
        
        # 提取新闻列表
        news_items = await context.query_selector_all('.news-item')
        
        for item in news_items:
            news = {
                "title": await item.query_selector('.title >> text'),
                "url": await item.query_selector('a.link >> @href'),
                "source": await item.query_selector('.source >> text'),
                "publish_time": await item.query_selector('.time >> text'),
                "crawl_time": datetime.now().isoformat(),
            }
            
            # 进入详情页采集正文和评论
            if news["url"]:
                await context.enqueue_links(selector='a.link')
            
            self.news_data.append(news)
        
        # 翻页处理
        next_page = await context.query_selector('.next-page')
        if next_page:
            await next_page.click()
            await context.wait_for_selector('.news-item')

async def main():
    crawler = FinanceNewsCrawler()
    await crawler.run([
        "https://finance.eastmoney.com/news/cjdp.html",
        "https://www.10jqka.com.cn/",
    ])
    
    # 保存数据
    await crawler.export_data('finance_news.csv')
    print(f"采集到 {len(crawler.news_data)} 条新闻")

if __name__ == '__main__':
    asyncio.run(main())
```

```python
        self.threshold = threshold  # 涨跌阈值
        self.alerts = []

    async def fetch_price(self, stock_code: str) -> dict:
        """获取单只股票价格"""
        crawler = BeautifulSoupCrawler(max_requests_per_crawl=1)

            # 模拟东方财富接口

            # 解析格式: var hq_str_sh600519="贵州茅台,..." 
        
        await crawler.run([url])
        
        # 简化示例：直接返回模拟数据
        return {"code": stock_code, "price": 1800.0, "change": 2.5}
```

### 4.1 完整数据流水线

```python
"""
量化数据工具体系 - 完整示例
从数据采集到分析输出的全流程
"""
import asyncio
from dataclasses import dataclass
from pathlib import Path
from typing import Optional
import json

# ==================== 配置 ====================
@dataclass
class QuantDataConfig:
    # Crawlee 配置
    crawler_user_agent: str = "Mozilla/5.0 (Windows NT 10.0; Win64; x64)..."
    crawler_proxy: Optional[str] = None
    crawler_timeout: int = 30
    
    # MinerU 配置
    mineru_backend: str = "pipeline"  # pipeline/vlm/hybrid
    mineru_language: str = "ch"  # ch/en/auto
    
    # 存储路径
    raw_data_dir: Path = Path("./data/raw")
    parsed_data_dir: Path = Path("./data/parsed")
    analysis_dir: Path = Path("./data/analysis")

class QuantDataPipeline:
    """
    量化数据流水线
    整合 Crawlee + MinerU + 分析技能
    """

    def __init__(self, config: QuantDataConfig):
        self.config = config
        self.config.raw_data_dir.mkdir(parents=True, exist_ok=True)
        self.config.parsed_data_dir.mkdir(parents=True, exist_ok=True)

    # ==================== 数据采集 ====================
    async def collect_news(self, keywords: list[str]) -> list[dict]:
        """采集财经新闻"""
        from crawlee.crawlers import BeautifulSoupCrawler
        
        crawler = BeautifulSoupCrawler()
        news_data = []

        @crawler.router.default_handler
        async def handler(context):
            items = context.soup.select('.news-item')
            for item in items:
                news_data.append({
                    "title": item.select_one('.title').text,
                    "url": item.select_one('a')['href'],
                    "time": item.select_one('.time').text,
                })
            await context.push_data(news_data)

        # 搜索财经新闻
        for kw in keywords:
            url = f"https://search.eastmoney.com/news?s={kw}"
            await crawler.run([url])

        return news_data

    async def collect_reports(self, stock_codes: list[str]) -> list[Path]:
        """
        采集券商研报 PDF
        注意: 实际使用需要对接研报平台 API 或购买数据
        """
        # 这里演示如何处理已下载的 PDF
        pdf_files = list(self.config.raw_data_dir.glob("**/*.pdf"))
        return pdf_files

    # ==================== 文档解析 ====================
    def parse_documents(self, pdf_files: list[Path]) -> list[dict]:
        """批量解析 PDF 文档"""
        from mineru import MinerU
        
        parser = MinerU(backend=self.config.mineru_backend)
        parsed_results = []

        for pdf_file in pdf_files:
            result = parser.parse(
                input_path=pdf_file,
                output_path=self.config.parsed_data_dir / pdf_file.stem,
                output_format="json",
                language=self.config.mineru_language
            )
            
            parsed_results.append({
                "file": str(pdf_file),
                "content": result.content,
                "tables": result.content.get("tables", []),
                "images": result.content.get("images", []),
            })

        return parsed_results

    def extract_financial_data(self, parsed_docs: list[dict]) -> dict:
        """
        从解析结果中提取财务数据
        对接已安装的财务分析技能
        """
        financial_data = {}
        
        for doc in parsed_docs:
            for table in doc.get("tables", []):
                # 识别表格类型
                table_text = str(table.get("title", ""))
                
                if "利润" in table_text or "损益" in table_text:
                    financial_data["income_statement"] = table["data"]
                elif "资产" in table_text or "负债" in table_text:
                    financial_data["balance_sheet"] = table["data"]
                elif "现金流" in table_text:
                    financial_data["cash_flow"] = table["data"]
                elif "预测" in table_text or "预计" in table_text:
                    financial_data["forecast"] = table["data"]

        return financial_data

    # ==================== 分析输出 ====================
    def generate_analysis(self, data: dict) -> str:
        """
        生成分析报告
        对接研报分析技能
        """
        analysis = f"""
# {data.get('title', '报告分析')} 自动分析

## 核心观点
{data.get('summary', '暂无')}

## 关键财务指标
{self._format_metrics(data.get('metrics', {}))}

## 风险提示
{data.get('risks', '暂无')}

## 投资建议
{data.get('recommendation', '暂无')}
"""
        return analysis

    def _format_metrics(self, metrics: dict) -> str:
        """格式化财务指标"""
        lines = []
        for key, value in metrics.items():
            lines.append(f"- {key}: {value}")
        return "\n".join(lines) if lines else "暂无数据"

# ==================== 使用示例 ====================
async def main():
    # 初始化配置
    config = QuantDataConfig(
        raw_data_dir=Path("./data/raw"),
        parsed_data_dir=Path("./data/parsed"),
    )
    
    # 创建流水线
    pipeline = QuantDataPipeline(config)
    
    # Step 1: 采集新闻
    print("1. 采集财经新闻...")
    news = await pipeline.collect_news(["宁德时代", "新能源车"])
    print(f"   采集到 {len(news)} 条新闻")
    
    # Step 2: 解析研报
    print("2. 解析研报文档...")
    pdf_files = await pipeline.collect_reports(["300750"])
    parsed = pipeline.parse_documents(pdf_files)
    print(f"   解析了 {len(parsed)} 份文档")
    
    # Step 3: 提取财务数据
    print("3. 提取财务数据...")
    financial_data = pipeline.extract_financial_data(parsed)
    print(f"   提取到 {len(financial_data)} 类财务数据")
    
    # Step 4: 生成分析
    print("4. 生成分析报告...")
    report = pipeline.generate_analysis(financial_data)
    print(report)

if __name__ == "__main__":
    asyncio.run(main())
```

## 七、常见问题与解决方案

### 7.1 MinerU 常见问题

| 问题 | 解决方案 |
|------|----------|
| **安装失败** | 确保 Python 3.10-3.13，使用 `uv pip install` |
| **GPU 不可用** | 设置 `backend="pipeline"` 使用 CPU 模式 |
| **表格识别不准** | 检查 PDF 清晰度，可尝试 OCR 模式 |
| **中文乱码** | 设置 `language="ch"` 或 `"auto"` |
| **长文档内存不足** | 启用滑动窗口机制，减少并发 |

### 7.2 Crawlee 常见问题

| 问题 | 解决方案 |
|------|----------|
| **被反爬拦截** | 启用代理轮换，降低请求频率 |
| **JS 渲染失败** | 使用 `PlaywrightCrawler` 而非 `BeautifulSoupCrawler` |
| **内存占用高** | 调整 `max_concurrency` 和 `max_pages_per_crawl` |
| **数据丢失** | 确保使用持久化存储而非内存 |
| **IP 被封** | 配置代理池，使用 `Session` 管理 |

### 7.3 性能优化建议

1. **MinerU 优化**
   - GPU 环境下启用 CUDA 加速
   - 批量处理时启用多线程
   - 长文档使用流式写入

2. **Crawlee 优化**
   - 静态页面使用 HTTP 爬虫
   - 动态页面才启用浏览器
   - 使用 `AdaptivePlaywrightCrawler` 自动选择

3. **存储优化**
   - 原始数据存入对象存储
   - 处理后数据存入数据库
   - 热点数据缓存到 Redis
